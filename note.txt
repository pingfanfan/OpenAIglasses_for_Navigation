Model

# 1) 盲道分割：Ultralytics YOLO11-L Seg 预训练
curl -L -o model/yolo-seg.pt https://github.com/ultralytics/assets/releases/download/v8.3.0/yolo11l-seg.pt

# 2) 开放词汇 YOLOE-11L-Seg
curl -L -o model/yoloe-11l-seg.pt https://huggingface.co/jameslahm/yoloe/resolve/main/yoloe-11l-seg.pt?download=true

# 3) 红绿灯检测：下载并解压，然后重命名为 trafficlight.pt
curl -L -o /tmp/tl.zip https://github.com/Satish-Vennapu/Traffic-Light-Detection-using-YOLOv8/raw/main/traffic_light_detection_trained_weights.zip
unzip -j /tmp/tl.zip -d model && mv model/best.pt model/trafficlight.pt

# 4) 手部检测（用你仓库里的现成文件）
curl -L -o model/hand_landmarker.task https://raw.githubusercontent.com/pingfanfan/OpenAIglasses_for_Navigation/main/hand_landmarker.task
